```c++
#include<iostream>
#include<cstdio>
#include<stack>
#include<queue>
using namespace std;
const int maxn = 10000+10;
int sz=0;

struct bitree{
	int v, l, r, p, h;
 	bitree(){
		p = l = r = -1;
		h = 0;
	}
}tr[maxn];

void updateheight(int rt) {
	tr[rt].h = max(tr[tr[rt].l].h, tr[tr[rt].r].h)+1;
}

void insert(int x, int rt, int p) {
	if (rt != -1) {
		if (x > tr[rt].v)
			insert(x, tr[rt].r, rt), updateheight(rt);
		else
			insert(x, tr[rt].l, rt), updateheight(rt);
  	}
 	else {
		tr[sz].v = x;
		if (tr[p].v > x)
			tr[p].l = sz;
		else
			tr[p].r = sz;
		tr[sz].p = p;
		sz++;
 	}
}


void travPre(int rt) {
	stack<int>s;
	while (true) {
		while (rt != -1) {
			printf("%d ", tr[rt].v);  //先查询根结点
			if (tr[rt].r != -1)
				s.push(tr[rt].r);	  //暂存右节点 
			rt = tr[rt].l;
		}
		if (s.empty())
			break;
		rt = s.top(), s.pop();        //弹出下一批的起点。
	}
}

void travIn(int rt) {
	stack<int>s;
	while (true) {
		if (rt != -1) {              //找最左节点
			s.push(rt);
			rt = tr[rt].l;
		}
		else if (!s.empty()) {       //根结点，之后转向该节点的右孩子
			rt = s.top();
			s.pop();
			printf("%d ", tr[rt].v);
			rt = tr[rt].r;
		}
		else
			break;
	}
}


void travPost(int rt) {
	stack<int>s;
	if (rt != -1)
		s.push(rt);
	while (!s.empty()) {
		int tmp = s.top();
		if (tmp != tr[rt].p) {      //若栈顶元素不是当前的节点的父亲，则为右孩子节点
			int x=s.top();
			while (x != -1) {       //在右子树中找最左的节点，记录右孩子。
				if (tr[x].l != -1) {
					if (tr[x].r != -1)
						s.push(tr[x].r);
					s.push(tr[x].l);
				}
				else
					s.push(tr[x].r);
				x = s.top();
			}
			s.pop();                //弹出空节点
		}
		rt = s.top();            
		s.pop();
		printf("%d ", tr[rt].v);
	}
}

int Find(int x, int rt) {           //查找
	int a;
	if (x == tr[rt].v)
		return rt;
	else if (x > tr[rt].v)
		a = Find(x, tr[rt].r);
	else
		a = Find(x, tr[rt].l);
	return a;
}

void travLevel(int rt) {            //层次遍历
	queue<int>q;
	q.push(rt);
	while (!q.empty()) {
		rt = q.front();
		q.pop();
		printf("%d ", tr[rt].v);
		if (tr[rt].l != -1)
			q.push(tr[rt].l);
		if (tr[rt].r != -1)
			q.push(tr[rt].r);	
	}
}

int main() {
	int n;
	scanf("%d", &n);
	for (int i=0; i<n; i++) {
		int x=0;
		scanf("%d", &x);
		if (i == 0)
			tr[0].v = x, sz++;
		else
			insert(x, 0, -1);
     	}
	
	cout<<endl;
	travPre(0);         //迭代先序遍历
	cout<<endl;
	travIn(0);          //迭代先序遍历
	cout<<endl;
	travPost(0);        //迭代后序遍历
	cout<<endl;
	travLevel(0);       //层次遍历
	cout<<Find(2, 0)<<endl;
	return 0;
}
```
